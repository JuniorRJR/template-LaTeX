% referencias
\begin{comment}

    bibjustif - justificar referências
    abnt-etal-text=it - texto (et.al.) em itálico
\end{comment}
\usepackage[alf, bibjustif, abnt-etal-text=it]{abntex2cite}

\usepackage[font=small]{caption}

% manutenção da marcação em listas (enumerator)
\usepackage{enumitem}

\usepackage[left=3.0cm, right=2.0cm, top=3.0cm, bottom=2.0cm]{geometry}

% inserir figuras
\usepackage{graphicx}

% hifenização das palavras
\usepackage{hyphenat}

\usepackage{indentfirst}

% acentuação direta
\usepackage[utf8]{inputenc}

% rotacionar elementos
\usepackage{rotating}

% manutenção do espaçamento entre linhas
\usepackage{setspace}
